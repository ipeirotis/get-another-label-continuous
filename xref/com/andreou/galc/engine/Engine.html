73  	  Double estimateDistributionSigma() {
74  
75  		Double nominator_sigma = 0.0;
76  		Double denominator_sigma = 0.0;
77  		for (Worker w : ip.getWorkers()) {
78  			Double b = w.getBeta();
79  			Double coef = Math.sqrt(b * b - b);
80  			Double s = w.getEst_sigma();
81  			nominator_sigma += coef * s;
82  			denominator_sigma += b;
83  		}
84  		Double est_sigma = nominator_sigma / denominator_sigma;
85  		return est_sigma;
86  	}
