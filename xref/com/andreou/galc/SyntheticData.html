85  	private void createLabels() {
86  
87  		// Generate Observation Values y_ij
88  
89  		for (DatumCont d : this.objects) {
90  			for (Worker w : this.workers) {
91  
92  				Double datum_z = (d.getTrueValue() - this.data_mu) / this.data_sigma;
93  				Double label_mu = w.getTrueMu() + w.getTrueRho() * w.getTrueSigma() * datum_z;
94  				Double label_sigma = Math.sqrt(1 - Math.pow(w.getTrueRho(), 2)) * w.getTrueSigma();
95  
96  				Generator labelGenerator = new Generator(Generator.Distribution.GAUSSIAN);
97  				labelGenerator.setGaussianParameters(label_mu, label_sigma);
98  				Double label = labelGenerator.nextData();
99  
100 				AssignedLabel al = new AssignedLabel(w.getName(), d.getName(), label);
101 				labels.add(al);
102 				w.addAssignedLabel(al);
103 				d.addAssignedLabel(al);
104 			}
105 		}
106 	}

108 	private void createObjects(int k_objects) {
109 
110 		// Generate Object Real Values x_i
111 		for (int i = 0; i < k_objects; i++) {
112 			DatumCont d = new DatumCont("Object" + (i + 1));
113 			Double v = datumGenerator.nextData();
114 			Double z = (v-this.data_mu)/this.data_sigma;
115 			d.setTrueValue(v);
116 			d.setTrueZeta(z);
117 			this.objects.add(d);
118 		}
119 	}
